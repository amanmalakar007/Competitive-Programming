class Solution {
    public static int reverse(int x)
	{
		long y=x,reverse=0,rem;
		int flag=0;
		String input=Long.toString(y);
		if(input.charAt(0)=='-')
		{
			input=input.substring(1);
			y=Long.parseLong(input);
			flag=1;
		}
		while(y>0)
		{
			rem=y%10;
			reverse=(reverse*10)+rem;
			if(reverse>2147483647 ||(reverse*-1<-2147483648))
				return 0;
			y=y/10;
		}
		if(flag==1)
			return (int)(reverse*-1);
		else if(flag==0)
			return (int)reverse;
		else
			return 0;
	}
}
